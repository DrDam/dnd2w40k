% #### et ##### en Markdown -> \subsection et \subsubsection
% (avec --top-level-division=part : # part, ## chapter, ### section,
% #### subsection, ##### subsubsection).
% Style demandé : pas de numéro, texte souligné (au lieu du gras seul
% par défaut), #### avec un filet plus épais que ##### pour marquer la
% hiérarchie (voir \subsectionrulewidth/\subsubsectionrulewidth plus
% bas). secnumdepth global reste à 5 (réglé par --number-sections côté
% pandoc) pour ne pas affecter la numérotation des autres niveaux --
% seul le format de \subsection et \subsubsection est changé, via titlesec.
%
% ATTENTION -- piège \underline et titres longs en mode twocolumn :
% \underline{...} (natif LaTeX2e, hérité du mode math) compose son
% contenu dans une \hbox -- une boîte RIGIDE, sans aucun point de
% coupure de ligne autorisé à l'intérieur. Un \subsubsection assez
% court tient dans la largeur de colonne par hasard, mais dès qu'un
% titre dépasse cette largeur (ex: "Modification sur une fiche
% personnage standard de D&D 5e."), la boîte ne peut PAS se replier sur
% plusieurs lignes : elle déborde tel quel par-dessus la colonne
% voisine, sans avertissement ni erreur de compilation (reproduit et
% confirmé en isolation, hors de tout pipeline Pandoc/Lua -- comportement
% de base de \underline, pas un artefact des filtres de ce projet).
% \soulpkg{soul} (\ul{...}) résout ce point nativement mais ajoute une
% dépendance non utilisée par ailleurs dans ce document.
%
% Deuxième piège -- \titlerule fait TOUJOURS pleine largeur de colonne :
% un premier correctif (remplacer \underline par \titlerule en 5e
% argument optionnel de \titleformat, APRÈS le titre) règle bien le
% débordement, mais \titlerule trace son filet sur \linewidth dans tous
% les cas, QUEL QUE SOIT le texte du titre au-dessus -- "Magie" (5
% lettres) se retrouvait donc souligné sur toute la largeur de la
% colonne, comme un titre 5x plus long (constaté en test, visuellement
% incohérent avec un soulignement "normal").
%
% Solution retenue (toujours sans nouveau package) : \titlerulewrap
% mesure (\settowidth) la largeur RÉELLE du titre tel que composé sur
% une seule ligne, et trace une règle de cette largeur précise -- sauf
% si cette largeur dépasse \linewidth (titre long qui se repliera de
% toute façon sur plusieurs lignes), auquel cas on retombe sur la
% pleine largeur de colonne, seul choix cohérent dans ce cas (le titre
% occupant alors plusieurs lignes, caler le filet sur la largeur d'une
% ligne précise -- ex: la dernière -- demanderait de mesurer une mise
% en page déjà figée, non disponible simplement sans dépendance
% supplémentaire telle que soul/ulem).
\usepackage{titlesec}
\usepackage{calc}

% --- Épaisseur des filets de titre (configurable) ---
% Même logique que le bloc de tailles de police plus haut : ces deux
% commandes sont le SEUL endroit à toucher pour ajuster l'épaisseur des
% traits sous les titres #### et #####. \subsectionrulewidth (####) est
% volontairement plus épais que \subsubsectionrulewidth (#####), pour
% marquer visuellement que #### est un niveau de titre au-dessus dans
% la hiérarchie -- tout le reste de leur traitement (gras, pas de
% numéro, largeur du trait calée sur le texte) reste identique entre
% les deux niveaux, voir \titlerulecore ci-dessous.
\newcommand{\subsubsectionrulewidth}{0.5pt}
\newcommand{\subsectionrulewidth}{1.5pt}

\newlength{\titleulwidth}

% Noyau commun : #1 = épaisseur du filet (une des deux commandes
% ci-dessus), #2 = texte du titre, injecté automatiquement par titlesec
% dans le "before-code" de \titleformat (voir plus bas). Factorisé ici
% pour que #### et ##### partagent exactement la même mesure de largeur
% et le même espacement filet/texte, seule l'épaisseur variant.
\newcommand{\titlerulecore}[2]{%
  \settowidth{\titleulwidth}{#2}%
  \ifdim\titleulwidth>\linewidth
    \setlength{\titleulwidth}{\linewidth}%
  \fi
  % \vspace{-0.6\baselineskip} (et non un \vspace positif) : sans ce
  % correctif, l'espace naturel laissé par \par après le titre (interligne
  % + profondeur de descente des caractères) crée un écart visuellement
  % disproportionné entre le texte du titre et le filet -- alors que
  % l'intention de départ (cf. l'ancien \underline) était un trait
  % "collé" au texte, façon soulignement classique. Valeur négative
  % choisie empiriquement pour coller le filet juste sous la ligne de
  % base du titre, quelle que soit sa taille de police.
  #2\par\nobreak\vspace{-1.25\baselineskip}\noindent\rule{\titleulwidth}{#1}%
}

% Wrappers à UN seul argument -- \titleformat n'injecte le titre composé
% que dans une macro à 1 argument (voir la 5e valeur des deux
% \titleformat ci-dessous) ; on ne peut donc pas passer l'épaisseur en
% argument à cet endroit-là, d'où ces deux wrappers minces, un par
% niveau de titre, qui referment \titlerulecore avec la bonne épaisseur.
\newcommand{\titlerulewrap}[1]{\titlerulecore{\subsubsectionrulewidth}{#1}}
\newcommand{\titlerulewrapThick}[1]{\titlerulecore{\subsectionrulewidth}{#1}}

\titleformat{\subsection}
  {\normalfont\bfseries}
  {}
  {0pt}
  {\titlerulewrapThick}

\titleformat{\subsubsection}
  {\normalfont\bfseries}
  {}
  {0pt}
  {\titlerulewrap}

% Espacement autour du titre de chapitre (avant / après / "indent" du
% bloc titre lui-même, dernier paramètre laissé à 0pt = pas de retrait).
% Par défaut, titlesec retombe sur des valeurs proches de book.cls natif
% (\vspace*{50pt} avant, \vskip 40pt après -- voir \@makechapterhead
% dans book.cls) dès que \titleformat est redéfini sans \titlespacing
% associée. Ces valeurs sont pensées pour un chapitre qui commence
% directement par du texte ; ici, le chapitre est souvent suivi d'une
% image (.wide), et le grand espace après-titre par défaut se voit
% comme un vide disproportionné au-dessus de l'image. Réduit ici à des
% valeurs plus resserrées, sans pour autant coller le titre au bord de
% page (le "avant" reste non nul pour respecter un minimum d'air en
% haut de page).
\titlespacing{\chapter}{0pt}{10pt}{10pt}

% ============================================================
% TABLE DES MATIÈRES EN DEUX COLONNES
% ============================================================
% Par défaut, la classe book bascule automatiquement \tableofcontents
% en une seule colonne pleine largeur (\onecolumn), même en mode
% twocolumn -- c'est un choix de design intégré à la classe standard,
% PAS un effet du document ou de pandoc. On redéfinit ici
% \tableofcontents pour retirer cette bascule et garder le mode deux
% colonnes du document.
%
% Point d'attention technique : il ne suffit pas de retirer le
% \onecolumn/\twocolumn de la définition d'origine -- le titre "Table
% des matières" lui-même utilise normalement \chapter*, qui pour tout
% titre de chapitre non numéroté en mode twocolumn déclenche un AUTRE
% mécanisme interne de la classe (\@topnewpage, via \@schapter) qui
% force le titre en pleine largeur de PAGE avant de basculer le
% contenu suivant en deux colonnes. Pour l'éviter complètement, le
% titre "Table des matières" est ici composé manuellement (mêmes
% styles visuels que \@makeschapterhead, mais sans passer par
% \chapter*/\@schapter), pour qu'il s'insère normalement dans le flux
% de la colonne courante comme n'importe quel autre titre.
\makeatletter
\renewcommand\tableofcontents{%
    \@mkboth{\MakeUppercase\contentsname}{\MakeUppercase\contentsname}%
    \@starttoc{toc}%
    }
\makeatother

% --- Sous-titre interne (Actions, Réactions, Traits légendaires...) ---
% Une fine ligne horizontale juste au-dessus du titre permet de
% détacher visuellement chaque section (Traits / Actions / ...) du
% texte qui précède, façon livre de règles -- sans quoi un sous-titre
% en simple gras se confondait avec le corps de texte environnant et
% pouvait sembler "ne pas s'afficher" même quand il était bien rendu.
\newcommand{\monstersection}[1]{%
  \par\vspace{6pt}
  \noindent\rule{\linewidth}{0.5pt}\par\vspace{3pt}
  {\bfseries\monstersectionfontsize #1}\par
  \vspace{2pt}
}

% --- Cadre sobre, façon livre de règles, sans bandeau coloré ---
% \tablefontsize (définie dans book/preamble.tex, = \small par défaut) :
% applique la même taille de police que les tableaux du document à TOUT
% le contenu du bloc monstre -- se propage automatiquement à monstername/
% monstersection/statline, qui sont émis à l'intérieur de ce groupe.
\newtcolorbox{monsterblock}{
  enhanced,
  breakable,
  colback=yellow!20,
  opacityback=0.95,
  colframe=black,
  boxrule=0.6pt,
  arc=1pt,
  left=8pt, right=8pt, top=6pt, bottom=6pt,
  before skip=12pt, after skip=12pt,
  fontupper=\tablefontsize,
}
